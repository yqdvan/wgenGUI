# wgen_GUI 配置文件示例
# 使用YAML语法指定Verilog模块名与文件路径的映射

# 示例1：简单模块
# simple_module: ./examples/simple_module.v

# 示例2：带参数的模块
# param_module: ./examples/param_module.v

# 示例3：复杂模块
# complex_module: ./examples/complex_module.v

modules:
  - module_name: simple_module
    ins_name: u_simple
    path: ./examples/simple_module.v
  - module_name: param_module
    ins_name: u_param_module
    path: ./examples/param_module.v
  - module_name: complex_module
    ins_name: u_complex
    path: ./examples/complex_module.v
  - module_name: axi4_fifo
    ins_name: u_axi4_fifo
    path: ./examples/axi4_fifo.v

generate_modules:
  - module_name: soc_chip
    path: ./examples/soc_chip.v
  - module_name: sublock
    path: ./examples/sublock.v    

hierarchy_def:
  - hierarchy: soc_chip
    includes:
      - u_simple
      - u_simple
      - u_param_module
      - u_complex
      - sublock
  - hierarchy: sublock
    includes:
      - u_complex
      - u_axi4_fifo
